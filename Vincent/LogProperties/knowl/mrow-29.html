\begin{align*} 4\log_{2}(x) + 3 \amp = \log_{2}\left(x^4\right) + 3 \tag{Power Rule}\\ \amp = \log_{2}\left(x^4\right) + \log_{2}\left(2^3\right) \tag{Since $3 = \log_{2}\left(2^3\right)$}\\ \amp = \log_{2}\left(x^4\right) + \log_{2}(8)\\ \amp = \log_{2}\left( 8x^4\right) \tag{Product Rule} \end{align*}
