\begin{align*} \ln \left(\dfrac{3}{ex}\right)^2 \amp = 2 \ln \left(\dfrac{3}{ex}\right) \tag{Power Rule}\\ \amp = 2 \left[ \ln(3) - \ln(ex) \right] \tag{Quotient Rule}\\ \amp = 2 \ln(3) - 2\ln(ex)\\ \amp = 2 \ln(3) - 2\left[\ln(e) + \ln(x)\right] \tag{Product Rule}\\ \amp = 2 \ln(3) - 2\ln(e) - 2 \ln(x)\\ \amp = 2\ln(3) - 2 - 2 \ln(x) \tag{Since $e^{1} = e$}\\ \amp = - 2 \ln(x) + 2\ln(3) - 2 \end{align*}
