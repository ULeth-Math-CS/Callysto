\begin{align*} -\ln(x) - \frac{1}{2} \amp = (-1)\ln(x) - \frac{1}{2}\\ \amp = \ln\left(x^{-1}\right) - \frac{1}{2} \tag{Power Rule}\\ \amp = \ln\left(x^{-1}\right) - \ln\left(e^{1/2}\right) \tag{Since $\frac{1}{2} = \ln\left(e^{1/2}\right)$}\\ \amp = \ln\left(x^{-1}\right) - \ln\left(\sqrt{e} \right)\\ \amp = \ln\left(\dfrac{x^{-1}}{\sqrt{e}}\right) \tag{Quotient Rule}\\ \amp = \ln\left(\dfrac{1}{x\sqrt{e}}\right) \end{align*}
